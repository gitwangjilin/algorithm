```

/**
 * 点
 */
public int value;
/**
 * 连接点

public int in;
/**
 * 连出点
 */
public int out;
/**
 * 下一个点
 */
public ArrayList<Node> nexts;
/**
 * 线
 */
public ArrayList<Edge> edges;
```
